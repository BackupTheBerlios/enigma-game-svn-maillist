+function drawmap(world, origin, resolver, ignore, map)
+    local len = string.len(ignore)
+    for y=1, #map do
+        local linelen = string.len(map[y])
+        for x = 1, linelen do
+            local key = string.sub(map[y], x, x)
+            if key ~= ignore then
+                world[origin + {x-1, y-1}] = resolver[key]
+            end
+        end
+    end
+end
